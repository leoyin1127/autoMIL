# Plan 02-07: Contract Test + BCK-04 Lint Script + Lint Pytest Gate

## Objective

Ship the Phase 2 correctness enforcement layer: (1) the parameterised contract test that proves the `Backend` ABC works against both `LocalBackend` and `MockSLURMBackend`, locking the ABC only after both pass; (2) the AST-walker lint script that enforces process-control isolation; (3) the pytest test that wraps the lint script as a mandatory CI gate.

Per D-70, the contract test runs ≥12 scenarios (S-01..S-12 from RESEARCH.md §3). Per D-64/D-65, the lint script is a plain stdlib `ast.NodeVisitor`, not ruff — allowlist is `backends/local.py` + `backends/_orchestrator_daemon.py`. The pytest gate is always-on enforcement; the pre-commit hook is optional convenience.

This is the highest-risk plan because it exercises both backends under real timing constraints (MockSLURM `poll_lag_seconds=0.05` with threading) and validates the AST walker against the actual production codebase.

Purpose: Without the contract test, the ABC is declared stable without proof against two implementations. Without the lint gate, BCK-04 process-isolation rules drift silently.

Output: `test_contract.py`, `check_backend_isolation.py`, `test_backend_isolation_lint.py`, updated `conftest.py`. Full test suite green including new tests.

    ```python
    import pytest
    import time
    from pathlib import Path
    from automil.backends.base import JobHandle, JobSpec, JobState

    @pytest.fixture(params=["local", "mock_slurm"])
    def backend(request, tmp_path):
        """Parameterised fixture: yields LocalBackend or MockSLURMBackend."""
        if request.param == "local":
            from automil.backends.local import LocalBackend
            # Create minimal project structure for LocalBackend
            automil_dir = tmp_path / "automil"
            (automil_dir / "orchestrator" / "queue").mkdir(parents=True)
            (automil_dir / "orchestrator" / "running").mkdir(parents=True)
            (automil_dir / "orchestrator" / "archive").mkdir(parents=True)
            (automil_dir / "config.yaml").write_text("backend:\n  name: local\n")
            # Minimal fake git repo
            (tmp_path / ".git").mkdir()
            yield LocalBackend(project_root=tmp_path, automil_dir=automil_dir)
        else:
            from automil.backends.mock_slurm import MockSLURMBackend
            yield MockSLURMBackend(poll_lag_seconds=0.05, state_file=tmp_path / "mock_state.json")

    def wait_for_state(backend, handle, target_states, timeout=5.0, interval=0.05):
        """Poll until job reaches any of target_states or raise TimeoutError."""
        deadline = time.monotonic() + timeout
        while time.monotonic() < deadline:
            state = backend.poll(handle)
            if state in target_states:
                return state
            time.sleep(interval)
        raise TimeoutError(
            f"Job {handle.opaque_id!r} did not reach {target_states} within {timeout}s; "
            f"last state: {backend.poll(handle)}"
        )

    def make_spec(node_id: str, tmp_path: Path, command=("echo", "hello"), **kwargs) -> JobSpec:
        """Build a minimal valid JobSpec for contract test scenarios."""
        overlay_dir = tmp_path / "archive" / node_id
        overlay_dir.mkdir(parents=True, exist_ok=True)
        return JobSpec(
            node_id=node_id,
            base_commit="abc1234",
            overlay_files=(),
            overlay_dir=overlay_dir,
            command=tuple(command),
            env=(),
            working_subdir=".",
            gpu_estimate_gb=0.5,
            walltime_seconds=3600,
            **kwargs,
        )
    ```

    ```python
    #!/usr/bin/env python3
    """Lint: forbid os.kill/os.killpg/os.getpid/Popen/.pid outside backend allowlist.

    BCK-04 / D-64: enforces process-control isolation.
    Run: python scripts/check_backend_isolation.py [src_root]
    Exit 0 if no violations; exit 1 with file:line diagnostics.
    """
    import ast
    import sys
    from pathlib import Path

    FORBIDDEN_NAMES = {"Popen"}
    FORBIDDEN_OS_ATTRS = {"kill", "killpg", "getpid"}
    FORBIDDEN_ATTR_END = "pid"
    # ALLOWLIST_PATHS rationale:
    # - backends/local.py + backends/_orchestrator_daemon.py — the ONLY job-control surfaces (D-60, D-64)
    # - viz/server.py — owns its OWN viz_server.pid lifecycle; its os.kill calls are a daemon-liveness probe
    #   (`os.kill(pid, 0)`) and SIGTERM stop, NOT job-control. Same category as _orchestrator_daemon.py's own
    #   PID file management. (Plan-check iter 1 B-04: lines 235/301/316 of viz/server.py would have failed the
    #   lint script on first run had this allowlist not been extended; the alternative — migrating viz daemon
    #   PID lifecycle to a backend — would be Phase 7 hardware-autodetect scope creep.)
    ALLOWLIST_PATHS = {
        Path("backends/local.py"),
        Path("backends/_orchestrator_daemon.py"),
        Path("viz/server.py"),
    }

    class BackendIsolationVisitor(ast.NodeVisitor):
        # ... (implement per RESEARCH.md §4 skeleton including _alias_map,
        #      visit_ImportFrom, visit_Name, visit_Attribute)
        # Critical: add visit_ImportFrom check for star-import of os/subprocess (Pitfall 4):
        #   if node.module in ("os", "subprocess") and any(a.name == "*" for a in node.names):
        #       self.violations.append((node.lineno, "forbidden: star-import of os/subprocess"))
        ...

    def main():
        src_root = Path(sys.argv[1]) if len(sys.argv) > 1 else Path("src/automil")
        violations = []
        for py_file in src_root.rglob("*.py"):
            rel = py_file.relative_to(src_root)
            if rel in ALLOWLIST_PATHS:
                continue
            try:
                tree = ast.parse(py_file.read_text(), filename=str(py_file))
            except SyntaxError as e:
                violations.append(f"{py_file}:0: SyntaxError: {e}")
                continue
            visitor = BackendIsolationVisitor(py_file, src_root)
            visitor.visit(tree)
            for lineno, msg in visitor.violations:
                violations.append(f"{py_file}:{lineno}: {msg}")
        if violations:
            print("VIOLATIONS:", file=sys.stderr)
            for v in violations:
                print(f"  {v}", file=sys.stderr)
            sys.exit(1)
        print("OK: no backend isolation violations")
        sys.exit(0)

    if __name__ == "__main__":
        main()
    ```
