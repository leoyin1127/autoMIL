    ```python
    class _RecordingBackend:
        def __init__(self):
            self.submitted_specs: list = []
            self._composites: dict[str, float] = {}  # cell_id -> composite
            self._states: dict[str, str] = {}        # job_id -> JobState string

        def set_composite(self, cell_id: str, composite: float):
            self._composites[cell_id] = composite

        def submit(self, spec):
            self.submitted_specs.append(spec)
            md = dict(spec.metadata)
            cell_id = md.get("cell_id")
            from automil.backends.base import JobHandle
            handle = JobHandle(job_id=spec.node_id, backend_name="mock", submitted_at=0)
            self._states[handle.job_id] = "completed"
            return handle

        def poll(self, handle):
            from automil.backends.base import JobState
            return JobState.COMPLETED  # deterministic for tests

        def cancel(self, handle, signal=None): pass
        def list_running(self): return []
        def log_iter(self, handle): yield ""
    ```

    ```python
    """Framework-purity guards for src/automil/gate/ (D-148 / Phase 4 analog).

    Mirrors tests/cells/ purity test if present. Asserts:
    - zero autobench/AUTOBENCH_/benchmarks/ refs (framework purity)
    - zero process-control refs (BCK-04 lint extension)
    - zero `git checkout` for rollback (Leo memory feedback_never_blind_checkout)
    """
    from __future__ import annotations
    import re
    from pathlib import Path
    import pytest

    GATE_DIR = Path("src/automil/gate")
    GATE_FILES = list(GATE_DIR.rglob("*.py"))

    def _read_all_gate() -> dict[Path, str]:
        return {p: p.read_text() for p in GATE_FILES}


    def test_gate_dir_has_files():
        assert GATE_FILES, f"No .py files found under {GATE_DIR}"


    def test_gate_no_autobench_refs():
        offenders = []
        for path, content in _read_all_gate().items():
            for token in ("autobench", "AUTOBENCH_", "benchmarks/"):
                if token in content:
                    offenders.append((path, token))
        assert not offenders, (
            f"Framework purity (D-148): {offenders} — gate/ must be autobench-free"
        )


    _PROCESS_CONTROL_RE = re.compile(
        r"\b(os\.kill|os\.killpg|os\.getpid|Popen|\.pid\b)"
    )


    def test_gate_no_process_control_refs():
        offenders = []
        for path, content in _read_all_gate().items():
            # Strip docstrings/comments? For simplicity, require literal absence.
            for line_no, line in enumerate(content.splitlines(), 1):
                stripped = line.lstrip()
                if stripped.startswith("#"):
                    continue
                if _PROCESS_CONTROL_RE.search(line):
                    offenders.append((path, line_no, line.strip()))
        assert not offenders, (
            f"BCK-04 lint extension: gate/ must not invoke process control. "
            f"Offenders: {offenders}"
        )


    def test_gate_no_blind_checkout():
        """Leo memory feedback_never_blind_checkout — never `git checkout` for rollback."""
        offenders = []
        for path, content in _read_all_gate().items():
            for forbidden in ('"git", "checkout"', "'git', 'checkout'", "git checkout --"):
                if forbidden in content:
                    offenders.append((path, forbidden))
        assert not offenders, (
            f"Leo memory feedback_never_blind_checkout: rollback must use path.unlink(), "
            f"NEVER `git checkout`. Offenders: {offenders}"
        )
    ```
