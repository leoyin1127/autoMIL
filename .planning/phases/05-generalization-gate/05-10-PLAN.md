    ```python
    async def promotion_rate_handler(request):
        """GTE-06 / D-144: serve current promotion_rate + diagnostic."""
        from automil.gate.stats import diagnose_gate_health
        from automil.graph import ExperimentGraph

        response_data = {
            "promotion_rate": 0.0,
            "nominated": 0,
            "promoted": 0,
            "health_diagnostic": "no data — no graph.json or zero nominations in window",
            "window_days": 30,
        }
        try:
            if GRAPH_FILE.exists():
                graph = ExperimentGraph(path=str(GRAPH_FILE))
                rate = graph.promotion_rate(days=30)
                nominated_nodes = graph.nominations_in_window(days=30)
                nominated_count = len(nominated_nodes)
                promoted_count = sum(
                    1 for n in nominated_nodes if n.get("status") == "registered"
                )
                response_data = {
                    "promotion_rate": rate,
                    "nominated": nominated_count,
                    "promoted": promoted_count,
                    "health_diagnostic": diagnose_gate_health(rate)
                                          if nominated_count > 0
                                          else "no data — zero nominations in 30-day window",
                    "window_days": 30,
                }
        except Exception:
            # Soft-fail: keep the default zero response (Pitfall 5a parallel discipline)
            pass
        return web.json_response(response_data)
    ```

    ```python
    import pytest
    from aiohttp.test_utils import TestClient, TestServer
    from aiohttp import web

    @pytest.mark.asyncio
    async def test_promotion_rate_endpoint_with_data(tmp_path, monkeypatch):
        adir = tmp_path / "automil"
        adir.mkdir()
        (adir / "config.yaml").write_text("")
        # Build a graph with 4 nominations, 2 registered
        from datetime import datetime, timezone
        ts = datetime.now(timezone.utc).isoformat()
        graph = {"nodes": {}}
        for i, status in enumerate(["registered", "registered", "keep", "keep"]):
            nid = f"node_{i:04d}"
            graph["nodes"][nid] = {
                "id": nid, "status": status,
                "history": [{"event": "nominated", "timestamp": ts, "agent_initiated": False}],
            }
        (adir / "graph.json").write_text(json.dumps(graph))
        monkeypatch.chdir(tmp_path)

        from automil.viz.server import create_app
        app = create_app()
        async with TestClient(TestServer(app)) as client:
            resp = await client.get("/api/promotion-rate")
            assert resp.status == 200
            data = await resp.json()
            assert data["promotion_rate"] == 0.5
            assert data["nominated"] == 4
            assert data["promoted"] == 2
            assert "healthy" in data["health_diagnostic"].lower()
            assert data["window_days"] == 30
    ```
