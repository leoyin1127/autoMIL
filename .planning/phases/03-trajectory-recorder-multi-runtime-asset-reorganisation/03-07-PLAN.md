  Implement `_install_runtime_assets(rt, project_root, package_dir, merge_skill)` as a helper inside `init.py`:
  ```python
  def _install_runtime_assets(rt: str, project_root, package_dir, merge_skill):
      """Install per-runtime overlay assets (skill, hooks, native files)."""
      from pathlib import Path as _Path
      import json as _json

      shared_dir = package_dir / "agent_assets" / "_shared"
      overlay_dir = package_dir / "agent_assets" / rt

      # Render SKILL.md for this runtime
      shared_skill = shared_dir / "skills" / "automil" / "SKILL.md"
      if shared_skill.exists():
          overlay_skill = overlay_dir / "SKILL.md" if (overlay_dir / "SKILL.md").exists() else None
          merged = merge_skill(rt, shared_skill, overlay_skill)
          # Write to automil/skills/<rt>/ in the project
          # (actual install location handled per runtime below)

      if rt == "claude":
          # .claude/CLAUDE.md starts with @AGENTS.md import
          project_claude = project_root / ".claude"
          project_claude.mkdir(exist_ok=True)
          claude_md = project_claude / "CLAUDE.md"
          if not claude_md.exists():
              claude_md.write_text("@AGENTS.md\n\n# Claude Code — autoMIL\n\nSee AGENTS.md above for universal instructions.\n")
              click.echo(f"  Created: .claude/CLAUDE.md")
          # Register hooks in settings.json
          _register_claude_hooks(project_root, project_claude, package_dir)
          click.echo(f"  Runtime: claude — assets installed")

      elif rt == "opencode":
          opencode_dir = project_root / ".opencode"
          opencode_dir.mkdir(exist_ok=True)
          # .opencode/AGENTS.md — same content as project-root AGENTS.md
          agents_src = package_dir / "agent_assets" / "_shared" / "AGENTS.md"
          if agents_src.exists():
              (opencode_dir / "AGENTS.md").write_text(agents_src.read_text(encoding="utf-8"))
          click.echo(f"  Runtime: opencode — assets installed (.opencode/AGENTS.md)")

      elif rt == "codex":
          codex_dir = project_root / ".codex"
          codex_dir.mkdir(exist_ok=True)
          # .codex/instructions.md
          instructions = codex_dir / "instructions.md"
          if not instructions.exists():
              agents_src = package_dir / "agent_assets" / "_shared" / "AGENTS.md"
              content = agents_src.read_text(encoding="utf-8") if agents_src.exists() else "# autoMIL\n\nSee AGENTS.md.\n"
              instructions.write_text(content)
          click.echo(f"  Runtime: codex — assets installed (.codex/instructions.md)")

      elif rt in ("deepseek-via-opencode", "deepseek-via-codex"):
          base_rt = "opencode" if "opencode" in rt else "codex"
          click.echo(f"  Runtime: {rt} — DeepSeek is a model; installing {base_rt} overlay")
          _install_runtime_assets(base_rt, project_root, package_dir, merge_skill)
  ```

      """Helper: initialise a bare git repo (copied from test_cli.py pattern)."""

      subprocess.run(["git", "init"], cwd=path, capture_output=True, check=True)
