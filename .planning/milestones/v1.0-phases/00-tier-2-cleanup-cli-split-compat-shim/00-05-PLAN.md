Replacement contract (D-04 verbatim):

```python
# Module-level constants (after existing constants block at orchestrator.py:30-40):
_SYSTEM_ENV_WHITELIST_LITERAL = frozenset({
    "PATH", "HOME", "USER", "SHELL", "LANG", "TZ", "TMPDIR",
    "LD_LIBRARY_PATH", "PYTHONPATH",
})
_SYSTEM_ENV_WHITELIST_PREFIX = ("LC_", "CUDA_", "NVIDIA_", "AUTOMIL_")

# New helper (private to ExperimentOrchestrator):
def _build_subprocess_env(self, *, gpu_id, node_id, archive, spec, pythonpath, worktree_benchmarks):
    env: dict[str, str] = {}
    # 1. System whitelist (literal + prefix-glob).
    for key, value in os.environ.items():
        if key in _SYSTEM_ENV_WHITELIST_LITERAL or key.startswith(_SYSTEM_ENV_WHITELIST_PREFIX):
            env[key] = value
    # 2. Config-driven passthrough (literal names only).
    for key in self._env_passthrough:  # set at __init__
        if key in os.environ:
            env[key] = os.environ[key]
    # 3. Orchestrator-injected (always overrides 1+2).
    env["CUDA_VISIBLE_DEVICES"] = str(gpu_id)
    env["AUTOMIL_GPU"] = "0"
    env["AUTOMIL_DESC"] = spec.get("description", "")
    env["AUTOMIL_NODE_ID"] = node_id
    env["AUTOMIL_RESULTS_DIR"] = str(archive.resolve())
    env["AUTOBENCH_ROOT"] = str(worktree_benchmarks.resolve())  # D-05: stays
    env["PYTHONPATH"] = pythonpath
    # 4. Per-spec env (last-write-wins).
    for k, v in spec.get("env", {}).items():
        if k not in ("AUTOMIL_GPU", "CUDA_VISIBLE_DEVICES"):
            env[k] = str(v)
    return env
```

    ```python
        self,
        *,
        gpu_id: int,
        node_id: str,
        archive: Path,
        spec: dict,
        pythonpath: str,
        worktree_benchmarks: Path,
    ) -> dict[str, str]:
        """Build the subprocess environment from a hardcoded whitelist + config passthrough.

        Replaces the previous `{**os.environ, ...}` leak. See CLN-02 / D-04.

        Layering (highest precedence wins):
          1. System whitelist (literal + prefix-glob match against os.environ).
          2. Config passthrough (literal names from automil/config.yaml: env.passthrough).
          3. Orchestrator-injected (always overrides 1+2).
          4. Per-spec spec.env (last-write-wins, except _SPEC_ENV_BLOCKED keys).
        """
        env: dict[str, str] = {}

        # 1. System whitelist
        for key, value in os.environ.items():
            if key in _SYSTEM_ENV_WHITELIST_LITERAL or key.startswith(_SYSTEM_ENV_WHITELIST_PREFIX):
                env[key] = value

        # 2. Config passthrough
        for key in self._env_passthrough:
            if key in os.environ:
                env[key] = os.environ[key]

        # 3. Orchestrator-injected
        env["CUDA_VISIBLE_DEVICES"] = str(gpu_id)
        env["AUTOMIL_GPU"] = "0"
        env["AUTOMIL_DESC"] = spec.get("description", "")
        env["AUTOMIL_NODE_ID"] = node_id
        env["AUTOMIL_RESULTS_DIR"] = str(archive.resolve())

        env["PYTHONPATH"] = pythonpath

        # 4. Per-spec env
        for k, v in spec.get("env", {}).items():
            if k not in _SPEC_ENV_BLOCKED:
                env[k] = str(v)

        return env
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Operator shell `os.environ` → experiment subprocess `env` | The orchestrator's parent process inherits secrets like `OPENAI_API_KEY`, `WANDB_API_KEY`, `GITHUB_TOKEN`, `AWS_SECRET_ACCESS_KEY`. A buggy or untrusted training script (the orchestrator runs ARBITRARY code via overlays) inherits the full env today and can exfiltrate via network calls, log lines, or `result.json`. **THIS IS THE HIGHEST-SEVERITY THREAT IN PHASE 0.** |
| `automil/config.yaml: env.passthrough` → orchestrator → subprocess | Operator-trusted; consumer-config-declared. But unbounded passthrough keys could re-introduce the leak. Mitigation: passthrough is literal-names-only (no globs at config layer per D-04). |
| Per-spec `spec.env` → subprocess | Agent-supplied; trusted-by-construction (the agent IS the orchestrator's child) but bounded by `_SPEC_ENV_BLOCKED` for `AUTOMIL_GPU` / `CUDA_VISIBLE_DEVICES` to prevent GPU-mask spoofing. |
