def test_add_executed_round_trips_arbitrary_metric_keys(empty_graph: ExperimentGraph):
    """D-200: dict-spread preserves all consumer keys (Pitfall 7b defender)."""

    assert n["metrics"]["test_bacc"] == 0.80
