<objective>
Ship the runtime (instantiate-time) identity validator. Per D-30, identity is the third validator in the chain — it runs in `train.py` BEFORE the first epoch, after the variant class has been resolved + instantiated. Per D-31, mode flag (`registry.mode: free | architecture-preserving`) modulates its strictness:

- **`free` mode (default):** validates only that the variant's output is compatible with the parent's loss expectations — dtype + tensor rank, NOT param count. Permits architectural exploration.
- **`architecture-preserving` mode:** adds parameter-count tolerance (configurable via `registry.identity_constraints` — e.g., `param_count_pct: 5` means variant param count within ±5% of parent) + per-layer shape audit. Locks the F1-classic identity-preserving search.

Per D-32, on failure: write `validation_failure.json` to the caller-supplied archive directory (`archive/<node_id>/`) + raise `ValidationError`. The training script's wrapper (consumer-side follow-up; full consumer train.py is post-Phase-1 per D-49) catches the exception and writes a `result.json` of `{status: "validation_failed", composite: 0.0}` so the orchestrator records the attempt without hanging.

Per Plan 01-01's torch-free invariant, torch is imported LAZILY inside the validator's `check()` method — `automil.registry.validators` remains importable without torch (matters for the synthetic mini-consumer in Plan 01-12).

Purpose: Without identity validation, a model variant with the wrong output shape would crash deep inside the loss computation 1-30 minutes into training, wasting GPU time. The instantiate-time stub-forward catches the smell on a 0.1-second tensor pass.

```python
from __future__ import annotations

import json
import logging
import os
import tempfile
from dataclasses import dataclass
from pathlib import Path
from typing import Any, Optional

from automil.registry.errors import ValidationError

logger = logging.getLogger(__name__)


@dataclass(frozen=True)
class IdentityConstraint:
    """Parsed form of one entry in `registry.identity_constraints`.

    Format examples:
      - "param_count_pct: 5" -> IdentityConstraint(kind="param_count_pct", value="5")
      - "output_rank: 2"     -> IdentityConstraint(kind="output_rank", value="2")
    """
    kind: str
    value: str


def _parse_constraints(raw: tuple[str, ...]) -> tuple[IdentityConstraint, ...]:
    """Parse `["k1: v1", "k2: v2"]` into IdentityConstraint tuples.

    Hard-fail (ValueError) on malformed entries with the offending entry named.
    """


class IdentityValidator:
    """Runtime stub-forward identity check.

    Args:
        mode: 'free' or 'architecture-preserving'.
        identity_constraints: tuple of "key: value" strings (parsed lazily).

    The check() method:
      1. Lazy-imports torch.
      2. Constructs minimal stub inputs (features=(1, feat_dim), coords=(1,2)).
      3. Runs variant.forward(features, coords) and parent.forward(features, coords)
         (parent_cls instantiated identically — both consume the same `args`).
      4. Compares output shape, dtype, rank.
      5. In 'architecture-preserving' mode: also compares parameter count +
         layer shapes per identity_constraints.
      6. On failure: writes archive_dir/validation_failure.json with structured
         expected-vs-actual + raises ValidationError.

    feat_dim defaults to 768 (a common ViT/CLAM-Small embed dim); the caller
    can override via `args.feat_dim` if available.
    """

    def __init__(
        self,
        mode: str = "free",
        identity_constraints: tuple[str, ...] = (),
    ) -> None:
        self.mode = mode
        self.constraints = _parse_constraints(identity_constraints)

    def check(
        self,
        variant_cls: type,
        parent_cls: type,
        archive_dir: Path,
        *,
        args: Optional[Any] = None,
        feat_dim: int = 768,
        n_instances: int = 4,
    ) -> None:
        """Run the runtime identity check. Raises ValidationError on failure.

        Args:
            variant_cls: the registered variant class (resolved from the registry).
            parent_cls: the parent baseline class (e.g., the framework's CLAM_MB).
                        For 'free' mode tests with no parent, callers can pass
                        variant_cls itself; the validator skips parent comparisons
                        in that case.
            archive_dir: the per-node archive directory; validation_failure.json
                         is written here on hard-fail (D-32).
            args: optional namespace passed to the variant constructor.
            feat_dim: stub feature dim; default 768.
            n_instances: stub bag size; default 4.
        """
```

    ```python
    @requires_torch
    def test_arch_preserving_param_count_happy(tmp_path):
        """4% more params; constraint `param_count_pct: 5` -> passes."""
        import torch
        from automil.registry import ModelVariant
        from automil.registry.validators.identity import IdentityValidator

        class Parent(ModelVariant):
            def __init__(self):
                # Fake "params" via a torch.nn.Module attribute. The validator
                # must use sum(p.numel() for p in module.parameters()) for the
                # count, which works on any nn.Module subclass. For these
                # tests we use a Linear layer with known param count.
                self.linear = torch.nn.Linear(10, 2)  # 10*2 + 2 = 22 params
            def forward(self, features, coords=None):
                return torch.zeros(1, 2)
            def parameters(self):
                yield from self.linear.parameters()

        class GoodVariant(ModelVariant):
            def __init__(self):
                # 22 + ~4% = 22.88, so 23 params (within 5%).
                # Use Linear(10, 2) with one extra unused buffer to bump count by 1.

            def forward(self, features, coords=None):
                return torch.zeros(1, 2)
            def parameters(self):
                yield from self.linear.parameters()
                yield self.extra
    ```

    ```python
    """Runtime identity validator (REG-03 identity, REG-06 mode / D-30 / D-31 / D-32).

    Runs at instantiate time in `train.py` BEFORE the first epoch:
      1. Lazy-imports torch.
      2. Constructs minimal stub inputs.
      3. Runs both variant.forward() and parent.forward() once.
      4. Compares output rank + dtype.
      5. (architecture-preserving only) compares param counts per
         identity_constraints + per-layer shape audit.
      6. On failure: writes archive_dir/validation_failure.json + raises
         ValidationError. The training script wrapper catches the exception
         and writes a result.json of {status: "validation_failed",
         composite: 0.0} so the orchestrator records the attempt.
    """
    from __future__ import annotations

    import json
    import logging
    import os
    import tempfile
    from dataclasses import dataclass
    from datetime import datetime, timezone
    from pathlib import Path
    from typing import Any, Optional

    from automil.registry.errors import ValidationError

    logger = logging.getLogger(__name__)


    @dataclass(frozen=True)
    class IdentityConstraint:
        """Parsed form of one entry in `registry.identity_constraints`."""
        kind: str
        value: str


    def _parse_constraints(raw: tuple[str, ...]) -> tuple[IdentityConstraint, ...]:
        out: list[IdentityConstraint] = []
        for entry in raw:
            if ":" not in entry:
                raise ValueError(
                    f"identity_constraints: malformed entry {entry!r}; "
                    f"expected 'kind: value' (e.g., 'param_count_pct: 5'). "
                    f"Edit automil/config.yaml: registry.identity_constraints."
                )
            k, _, v = entry.partition(":")
            out.append(IdentityConstraint(kind=k.strip(), value=v.strip()))
        return tuple(out)


    def _atomic_write_json(path: Path, data: dict) -> None:
        """PATTERNS.md §3 atomic write: tempfile.mkstemp + os.rename."""
        path.parent.mkdir(parents=True, exist_ok=True)
        tmp_fd, tmp_path = tempfile.mkstemp(dir=str(path.parent), suffix=".tmp")
        try:
            with os.fdopen(tmp_fd, "w") as f:
                json.dump(data, f, indent=2)
                f.write("\n")
            os.rename(tmp_path, str(path))
            os.utime(str(path))
        except Exception:
            if os.path.exists(tmp_path):
                os.unlink(tmp_path)
            raise


    def _safe_param_count(obj: Any) -> int:
        """Return number of parameter elements; 0 if obj has no .parameters()."""
        try:
            return sum(p.numel() for p in obj.parameters())
        except (AttributeError, TypeError):
            return 0


    class IdentityValidator:
        """Runtime stub-forward identity check.

        Mode-aware:
          - 'free': dtype + rank only.
          - 'architecture-preserving': adds param_count + per-constraint check.
        """

        def __init__(
            self,
            mode: str = "free",
            identity_constraints: tuple[str, ...] = (),
        ) -> None:
            if mode not in ("free", "architecture-preserving"):
                raise ValueError(
                    f"IdentityValidator: mode must be 'free' or "
                    f"'architecture-preserving'; got {mode!r}."
                )
            self.mode = mode
            self.constraints = _parse_constraints(identity_constraints)

        def check(
            self,
            variant_cls: type,
            parent_cls: type,
            archive_dir: Path,
            *,
            args: Optional[Any] = None,
            feat_dim: int = 768,
            n_instances: int = 4,
        ) -> None:
            """Run identity check. Hard-fail on mismatch."""
            # Lazy torch import — Plan 01-01 invariant: framework loadable without torch.
            try:
                import torch  # noqa: F401
            except ImportError as e:
                raise ValidationError(
                    validator_name="identity",
                    path=archive_dir,
                    reason="torch not installed; identity validator requires torch at runtime",
                    fix_suggestion="`pip install torch` or `pip install -e .[torch]`.",
                ) from e

            # Instantiate both classes. We pass `args` if provided; else no-arg constructor.
            try:
                variant_inst = variant_cls(args) if args is not None else variant_cls()
            except TypeError:
                # Some classes take no args at all; retry without.
                variant_inst = variant_cls()
            try:
                parent_inst = parent_cls(args) if args is not None else parent_cls()
            except TypeError:
                parent_inst = parent_cls()

            # Build stub inputs.
            features = torch.zeros(n_instances, feat_dim)
            coords = torch.zeros(n_instances, 2)

            # Run forwards. Catch ANY exception and surface as identity failure.
            try:
                v_out = variant_inst.forward(features, coords)
            except Exception as e:
                self._fail(
                    archive_dir,
                    variant_cls=variant_cls, parent_cls=parent_cls,
                    reason=f"variant.forward() raised: {type(e).__name__}: {e}",
                    fix_suggestion="Check the variant's forward() implementation.",
                )

            try:
                p_out = parent_inst.forward(features, coords)
            except Exception as e:
                logger.warning(
                    "Identity validator: parent.forward() raised %s: %s — comparing "
                    "variant against itself (degraded check).",
                    type(e).__name__, e,
                )
                p_out = v_out  # degrade gracefully — still validates variant produces tensor

            # Convert to torch.Tensor for inspection. ModelVariant.forward may return
            # tuples (CLAM-shape) — extract first tensor for rank/dtype check.
            v_tensor = self._extract_tensor(v_out)
            p_tensor = self._extract_tensor(p_out)

            if v_tensor is None:
                self._fail(
                    archive_dir,
                    variant_cls=variant_cls, parent_cls=parent_cls,
                    reason="variant.forward() returned no tensor",
                    fix_suggestion="Ensure forward returns at least one torch.Tensor.",
                )

            # Rank check.
            v_rank = v_tensor.dim()
            p_rank = p_tensor.dim() if p_tensor is not None else v_rank
            if v_rank != p_rank:
                self._fail(
                    archive_dir,
                    variant_cls=variant_cls, parent_cls=parent_cls,
                    reason=f"output rank mismatch: variant={v_rank}, parent={p_rank}",
                    fix_suggestion=(
                        f"Adjust variant.forward to return a rank-{p_rank} tensor."
                    ),
                    expected={"rank": p_rank, "dtype": str(p_tensor.dtype)},
                    actual={"rank": v_rank, "dtype": str(v_tensor.dtype)},
                )

            # Dtype check.
            if p_tensor is not None and v_tensor.dtype != p_tensor.dtype:
                self._fail(
                    archive_dir,
                    variant_cls=variant_cls, parent_cls=parent_cls,
                    reason=f"output dtype mismatch: variant={v_tensor.dtype}, parent={p_tensor.dtype}",
                    fix_suggestion=(
                        f"Cast variant output to {p_tensor.dtype} before return."
                    ),
                    expected={"rank": p_rank, "dtype": str(p_tensor.dtype)},
                    actual={"rank": v_rank, "dtype": str(v_tensor.dtype)},
                )

            # Architecture-preserving mode: param_count + per-constraint.
            if self.mode == "architecture-preserving" and self.constraints:
                self._run_constraints(
                    archive_dir, variant_cls, parent_cls,
                    variant_inst, parent_inst, v_tensor, p_tensor,
                )

        def _extract_tensor(self, output: Any) -> Optional[Any]:
            """ModelVariant.forward may return tuple/dict/single — return the first
            torch.Tensor we find."""
            import torch
            if isinstance(output, torch.Tensor):
                return output
            if isinstance(output, (list, tuple)):
                for item in output:
                    if isinstance(item, torch.Tensor):
                        return item
            if isinstance(output, dict):
                for v in output.values():
                    if isinstance(v, torch.Tensor):
                        return v
            return None

        def _run_constraints(
            self,
            archive_dir: Path,
            variant_cls: type,
            parent_cls: type,
            variant_inst: Any,
            parent_inst: Any,
            v_tensor: Any,
            p_tensor: Any,
        ) -> None:
            for c in self.constraints:
                if c.kind == "param_count_pct":
                    try:
                        tol_pct = float(c.value)
                    except ValueError:
                        raise ValueError(
                            f"identity_constraints: param_count_pct value "
                            f"{c.value!r} not a number; expected 'param_count_pct: 5'."
                        )
                    v_n = _safe_param_count(variant_inst)
                    p_n = _safe_param_count(parent_inst)
                    if p_n == 0:
                        # Can't compute %; skip with log.
                        logger.warning(
                            "param_count_pct: parent has 0 parameters; skipping check."
                        )
                        continue
                    pct = abs(v_n - p_n) / p_n * 100.0
                    if pct > tol_pct:
                        self._fail(
                            archive_dir,
                            variant_cls=variant_cls, parent_cls=parent_cls,
                            reason=(
                                f"param_count_pct exceeded: variant={v_n}, parent={p_n}, "
                                f"diff={pct:.2f}% > tolerance={tol_pct:.2f}%"
                            ),
                            fix_suggestion=(
                                f"Reduce variant param count or relax `param_count_pct` "
                                f"in registry.identity_constraints (currently {tol_pct})."
                            ),
                            expected={"param_count_pct_max": tol_pct},
                            actual={"param_count_pct": pct},
                        )
                # Other constraint kinds can be added later (output_rank, layer_shape, etc.).
                # Phase 1 ships param_count_pct; future constraints land in Phase 5.

        def _fail(
            self,
            archive_dir: Path,
            *,
            variant_cls: type,
            parent_cls: type,
            reason: str,
            fix_suggestion: str,
            expected: Optional[dict] = None,
            actual: Optional[dict] = None,
        ) -> None:
            """Atomic-write validation_failure.json + raise ValidationError."""
            report = {
                "validator_name": "identity",
                "mode": self.mode,
                "variant_class": variant_cls.__name__,
                "parent_class": parent_cls.__name__,
                "reason": reason,
                "expected": expected or {},
                "actual": actual or {},
                "constraints_evaluated": [
                    f"{c.kind}: {c.value}" for c in self.constraints
                ],
                "timestamp": datetime.now(tz=timezone.utc).isoformat(),
            }
            try:
                _atomic_write_json(archive_dir / "validation_failure.json", report)
            except Exception as e:
                logger.error("Could not write validation_failure.json: %s", e)
            raise ValidationError(
                validator_name="identity",
                path=archive_dir / "validation_failure.json",
                reason=reason,
                fix_suggestion=fix_suggestion,
            )
    ```
