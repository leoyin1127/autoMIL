```python
from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path
from typing import Optional


@dataclass(frozen=True)
class ScanResult:
    """Outcome of scan_variants(variants_root).

    `imported` is the list of variant module paths whose import succeeded
    (and thus whose @register decorator ran).

    `failed` is the list of (path, exception) tuples for modules whose import
    raised. Plan 01-09 (refresh-registry) decides whether to fail (--strict)
    or warn-and-continue based on this list.

    `skipped` is the list of paths the scanner intentionally skipped
    (`__init__.py`, `_*.py` other than the kind subdirs themselves).
    """
    imported: tuple[Path, ...] = field(default_factory=tuple)
    failed: tuple[tuple[Path, str], ...] = field(default_factory=tuple)  # (path, "ExcType: msg")
    skipped: tuple[Path, ...] = field(default_factory=tuple)


def scan_variants(variants_root: Path) -> ScanResult:
    """Walk variants_root and import every variant module.

    Discovery rules (D-29):
      - Each immediate subdirectory of variants_root is either a parent name
        ('clam_mb', 'ab_mil') OR one of '_losses' / '_policies' / '_candidates'.
      - Inside each subdir, every `*.py` file (other than `__init__.py` and
        any `_<name>.py` "private helper" files) is imported.
      - The scanner does NOT clear the registry beforehand; callers (Plan 01-09)
        call `_clear_registry()` first if they want a fresh start.

    Returns a ScanResult capturing imported / failed / skipped paths.

    Importing a module that contains `@register(VariantSpec(...))` triggers
    Plan 01-02's decorator, which validates the spec and inserts into the
    singleton dicts. A duplicate-name error from @register surfaces here as
    a failed-import entry.
    """


def regenerate_init_py(kind_dir: Path) -> str:
    """Regenerate `<kind_dir>/__init__.py` with imports-only manifest (D-29).

    The output looks like:
        # AUTO-GENERATED by `automil refresh-registry` at 2026-05-02T10:00:00Z. DO NOT EDIT.
        # Re-run the command after adding or renaming a variant module.
        from . import clam_mb_v0176  # noqa: F401
        from . import clam_mb_v0188  # noqa: F401

    Deterministic ordering: alphabetic on `module_name`.

    Idempotent: regenerating produces byte-identical output IF the directory
    contents haven't changed (the timestamp comment is on a SEPARATE line so
    the body bytes match across runs even if mtime advances).

    Returns the rendered content as a string AND writes it atomically to disk.
    """
```

```python
    """
    spec: VariantSpec

    source_overlay_files: tuple[str, ...]             # paths the dirty diff touched
    ported_at: str                                    # ISO-8601 UTC

    def write(self, path: Path) -> None:
        """Atomic-write the manifest as JSON to `path`. PATTERNS.md §3."""

        """Load + validate a manifest from disk. Raises ValueError on schema violation."""

        """Verify the manifest's spec matches the docstring header in module_path.

        Returns (ok, reason_if_not_ok). Plan 01-08 (`automil check`) calls this
        for every registered variant — mismatch is reported as a check failure.
        """
```

    ```python
    """Variant directory scanner + __init__.py regenerator (REG-02 / D-29)."""
    from __future__ import annotations

    import importlib
    import importlib.util
    import logging
    import os
    import sys
    import tempfile
    from dataclasses import dataclass, field
    from datetime import datetime, timezone
    from pathlib import Path

    logger = logging.getLogger(__name__)


    @dataclass(frozen=True)
    class ScanResult:
        imported: tuple[Path, ...] = field(default_factory=tuple)
        failed: tuple[tuple[Path, str], ...] = field(default_factory=tuple)
        skipped: tuple[Path, ...] = field(default_factory=tuple)


    def _import_path(path: Path) -> None:
        """Import a single .py file by absolute path.

        Uses a unique module name per call so re-scans don't pollute sys.modules.
        Re-importing the same path overwrites the previous module entry.
        """
        # Stable module name: full posix path with non-identifier chars replaced.
        # Using path.stem alone causes collisions when two parents have the same
        # variant name (e.g., variants/a/b.py and variants/c/b.py).
        safe = "_".join(p for p in path.parts).replace("/", "_").replace(".", "_")
        module_name = f"_automil_variant_{safe}"

        spec = importlib.util.spec_from_file_location(module_name, path)
        if spec is None or spec.loader is None:
            raise ImportError(f"cannot create module spec for {path}")
        module = importlib.util.module_from_spec(spec)
        sys.modules[module_name] = module  # required for relative imports inside the module
        try:
            spec.loader.exec_module(module)  # type: ignore[union-attr]
        except Exception:
            sys.modules.pop(module_name, None)
            raise


    def _is_variant_module(path: Path) -> bool:
        """Return whether path is a variant module (excludes __init__.py and _*.py helpers)."""
        if path.suffix != ".py":
            return False
        if path.name == "__init__.py":
            return False
        if path.name.startswith("_"):
            return False
        return True


    def scan_variants(variants_root: Path) -> ScanResult:
        """Walk variants_root and import every variant module."""
        if not variants_root.exists():
            return ScanResult()

        imported: list[Path] = []
        failed: list[tuple[Path, str]] = []
        skipped: list[Path] = []

        for sub in sorted(variants_root.iterdir()):
            if not sub.is_dir():
                continue
            for py_file in sorted(sub.iterdir()):
                if py_file.is_dir():
                    # No nested variant dirs in Phase 1; skip silently.
                    continue
                if not _is_variant_module(py_file):
                    if py_file.suffix == ".py":
                        skipped.append(py_file)
                    continue
                try:
                    _import_path(py_file)
                    imported.append(py_file)
                except Exception as e:
                    failed.append((py_file, f"{type(e).__name__}: {e}"))

        return ScanResult(
            imported=tuple(imported),
            failed=tuple(failed),
            skipped=tuple(skipped),
        )


    def regenerate_init_py(kind_dir: Path) -> str:
        """Regenerate kind_dir/__init__.py with imports-only manifest (D-29)."""
        if not kind_dir.exists():
            kind_dir.mkdir(parents=True, exist_ok=True)

        # Discover variant modules in alphabetic order.
        modules: list[str] = []
        for py_file in sorted(kind_dir.iterdir()):
            if py_file.is_file() and _is_variant_module(py_file):
                modules.append(py_file.stem)

        # Build the rendered content. Timestamp goes on a SEPARATE line so the
        # body bytes match across runs even if mtime advances (D-29 idempotent).
        ts = datetime.now(tz=timezone.utc).isoformat()
        header = (
            "# AUTO-GENERATED by `automil refresh-registry`. DO NOT EDIT.\n"
            "# Re-run the command after adding or renaming a variant module.\n"
            f"# generated-at: {ts}\n"
        )
        if modules:
            body = "\n".join(f"from . import {m}  # noqa: F401" for m in modules) + "\n"
        else:
            body = ""
        rendered = header + body

        # Atomic write (PATTERNS.md §3).
        target = kind_dir / "__init__.py"
        target.parent.mkdir(parents=True, exist_ok=True)
        tmp_fd, tmp_path = tempfile.mkstemp(dir=str(target.parent), suffix=".tmp")
        try:
            with os.fdopen(tmp_fd, "w") as f:
                f.write(rendered)
            os.rename(tmp_path, str(target))
            os.utime(str(target))
        except Exception:
            if os.path.exists(tmp_path):
                os.unlink(tmp_path)
            raise

        return rendered
    ```

    ```python
    """Sibling JSON manifest for registered variants (REG-08 / D-44)."""
    from __future__ import annotations

    import json
    import logging
    import os
    import re
    import tempfile
    from dataclasses import asdict, dataclass, field
    from pathlib import Path
    from typing import Any

    from automil.registry.spec import VariantSpec

    logger = logging.getLogger(__name__)


    @dataclass(frozen=True)
    class Manifest:
        spec: VariantSpec
        source_node: str
        source_overlay_files: tuple[str, ...]
        ported_at: str
        tool_version: str

        def write(self, path: Path) -> None:
            """Atomic-write the manifest as JSON. PATTERNS.md §3."""
            payload = {
                "spec": {
                    "name": self.spec.name,
                    "kind": self.spec.kind,
                    "parent": self.spec.parent,
                    "base_commit": self.spec.base_commit,
                    "composite": self.spec.composite,
                    "node_id": self.spec.node_id,
                    "created_at": self.spec.created_at,
                    "mutations": list(self.spec.mutations),
                },
                "source_node": self.source_node,
                "source_overlay_files": list(self.source_overlay_files),
                "ported_at": self.ported_at,
                "tool_version": self.tool_version,
            }
            path.parent.mkdir(parents=True, exist_ok=True)
            tmp_fd, tmp_path = tempfile.mkstemp(dir=str(path.parent), suffix=".tmp")
            try:
                with os.fdopen(tmp_fd, "w") as f:
                    json.dump(payload, f, indent=2)
                    f.write("\n")
                os.rename(tmp_path, str(path))
                os.utime(str(path))
            except Exception:
                if os.path.exists(tmp_path):
                    os.unlink(tmp_path)
                raise

        @classmethod
        def read(cls, path: Path) -> "Manifest":
            if not path.exists():
                raise FileNotFoundError(f"manifest not found: {path}")
            try:
                payload = json.loads(path.read_text())
            except json.JSONDecodeError as e:
                raise ValueError(
                    f"manifest at {path} is not valid JSON: {e.msg} (line {e.lineno})"
                ) from e

            for key in ("spec", "source_node", "source_overlay_files", "ported_at", "tool_version"):
                if key not in payload:
                    raise ValueError(f"manifest at {path} missing required key: {key!r}")

            spec_data = payload["spec"]
            for k in ("name", "kind", "parent", "base_commit", "composite", "node_id", "created_at"):
                if k not in spec_data:
                    raise ValueError(f"manifest at {path}: spec missing key {k!r}")

            spec = VariantSpec(
                name=spec_data["name"],
                kind=spec_data["kind"],
                parent=spec_data["parent"],
                base_commit=spec_data["base_commit"],
                composite=float(spec_data["composite"]),
                node_id=spec_data["node_id"],
                created_at=spec_data["created_at"],
                mutations=tuple(spec_data.get("mutations", [])),
            )

            return cls(
                spec=spec,
                source_node=payload["source_node"],
                source_overlay_files=tuple(payload["source_overlay_files"]),
                ported_at=payload["ported_at"],
                tool_version=payload["tool_version"],
            )

        def cross_check_with_module(self, module_path: Path) -> tuple[bool, str]:
            """Verify the docstring header matches the manifest spec (D-44)."""
            if not module_path.exists():
                return False, f"module not found: {module_path}"
            text = module_path.read_text()
            # Extract module-level docstring.
            doc = self._extract_docstring(text)
            if doc is None:
                return False, "no module docstring found"

            # Header is the first line: `<name> variant.` (allow .description).
            first_line = doc.strip().splitlines()[0] if doc.strip() else ""
            # Match e.g., "clam_mb_v0176 variant." or "clam_mb_v0176: <description>"
            name_match = re.match(r"^(\S+)", first_line)
            doc_name = name_match.group(1).rstrip(":") if name_match else None
            if doc_name and doc_name != self.spec.name:
                return False, (
                    f"docstring name {doc_name!r} != manifest name {self.spec.name!r}"
                )

            # Field-line parsing: "Parent: <x>", "Composite: <f>", etc.
            fields = self._parse_doc_fields(doc)

            if self.spec.kind == "model":
                doc_parent = fields.get("parent")
                if doc_parent and doc_parent != self.spec.parent:
                    return False, (
                        f"docstring parent {doc_parent!r} != manifest parent "
                        f"{self.spec.parent!r}"
                    )

            doc_composite = fields.get("composite")
            if doc_composite is not None:
                try:
                    doc_c = float(doc_composite)
                    if doc_c != self.spec.composite:
                        return False, (
                            f"docstring composite {doc_c} != manifest composite "
                            f"{self.spec.composite}"
                        )
                except ValueError:
                    return False, f"docstring composite not a float: {doc_composite!r}"

            doc_node = fields.get("node id") or fields.get("node_id")
            if doc_node and doc_node != self.spec.node_id:
                return False, (
                    f"docstring node_id {doc_node!r} != manifest node_id "
                    f"{self.spec.node_id!r}"
                )

            return True, ""

        def _extract_docstring(self, text: str) -> str | None:
            """Return the module docstring (None if absent)."""
            import ast
            try:
                tree = ast.parse(text)
            except SyntaxError:
                return None
            return ast.get_docstring(tree)

        def _parse_doc_fields(self, doc: str) -> dict[str, str]:
            """Parse `Key: value` lines into a lowercase-key dict."""
            fields: dict[str, str] = {}
            for line in doc.splitlines():
                if ":" not in line:
                    continue
                k, _, v = line.partition(":")
                fields[k.strip().lower()] = v.strip()
            return fields
    ```
