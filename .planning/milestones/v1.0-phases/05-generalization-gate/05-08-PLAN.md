    Create `src/automil/cli/gate.py`:
    ```python
    """gate subgroup: gate manifest + promotion-rate commands (GTE-01..06 / D-145)."""
    from __future__ import annotations

    import json
    import re
    from datetime import datetime, timezone
    from pathlib import Path

    import click

    from automil.cli import main


    @main.group("gate")
    def gate_group() -> None:
        """Generalization-gate manifest management commands."""
        pass


    _PARENT_ID_RE = re.compile(r"^node_\d+$")


    def _validate_parent_id(parent_id: str) -> None:
        if not _PARENT_ID_RE.match(parent_id):
            raise click.ClickException(
                f"Invalid parent_id {parent_id!r}: must match ^node_\\d+$ "
                f"(path-traversal defense)"
            )


    def _parse_held_out_cells(spec: str) -> tuple:
        """Parse --held-out-cells "id1:dataset:encoder:task,id2:..." into manifest tuple."""
        cells = []
        for entry in spec.split(","):
            parts = entry.strip().split(":")
            if len(parts) != 4:
                raise click.ClickException(
                    f"--held-out-cells entry {entry!r}: expected 'cell_id:dataset:encoder:task'"
                )
            cells.append(tuple(p.strip() for p in parts))
        return tuple(cells)


    @gate_group.command("register-manifest")
    @click.argument("parent_id")
    @click.option(
        "--strategy",
        type=click.Choice(["stratified", "random", "operator-curated"], case_sensitive=False),
        default="stratified",
        help="Held-out cell selection strategy (D-145, O-03). Stratified is recommended.",
    )
    @click.option("--K", "K", type=int, default=2, help="Min cells that must pass (D-141).")
    @click.option("--p-threshold", type=float, default=0.05, help="Pre-Bonferroni alpha (O-02).")
    @click.option("--bootstrap-reps", type=int, default=1000, help="GTE-04 locked: 1000.")
    @click.option(
        "--held-out-cells",
        default=None,
        help="Comma-separated 'cell_id:dataset:encoder:task' tuples. Required for operator-curated.",
    )
    @click.option(
        "--auto-select",
        type=int,
        default=None,
        help="For stratified/random: number of cells to auto-select.",
    )
    def register_manifest_cmd(
        parent_id: str,
        strategy: str,
        K: int,
        p_threshold: float,
        bootstrap_reps: int,
        held_out_cells: str | None,
        auto_select: int | None,
    ) -> None:
        """Register a pre-registered gate manifest for parent_id (D-138).

        Validates inputs, atomically writes manifest, commits to git in one operation.
        Refuses to overwrite an existing manifest — use retire-manifest first.
        """
        from automil.cli._helpers import _find_automil_dir, _find_git_root  # lazy
        from automil.gate import (
            GateManifest, write_manifest_committed, validate_manifest_dict,
        )
        from automil.gate.manifest import SCHEMA_VERSION  # lazy

        _validate_parent_id(parent_id)

        adir = _find_automil_dir()
        git_root = _find_git_root()

        # Validate parent_id is real graph node
        graph_path = adir / "graph.json"
        if graph_path.exists():
            data = json.loads(graph_path.read_text())
            if parent_id not in data.get("nodes", {}):
                raise click.ClickException(
                    f"parent_id {parent_id!r} not found in {graph_path}; "
                    f"submit/run the parent before registering its manifest"
                )

        # Resolve held-out cells
        if strategy == "operator-curated":
            if not held_out_cells:
                raise click.ClickException(
                    "--strategy operator-curated requires --held-out-cells "
                    "'cell_id1:ds:enc:task,cell_id2:...'"
                )
            cells = _parse_held_out_cells(held_out_cells)
        else:
            if held_out_cells:
                cells = _parse_held_out_cells(held_out_cells)
            elif auto_select:
                # Auto-selection deferred — stub points operator at explicit list.
                # Calibration pilot (plan 12) will populate the recipe; for now
                # the CLI surface ships and operator passes --held-out-cells.
                raise click.ClickException(
                    f"--strategy {strategy} --auto-select {auto_select}: "
                    f"automatic selection helper is forthcoming (calibration pilot, plan 12). "
                    f"Use --held-out-cells with an explicit comma-separated list for now."
                )
            else:
                raise click.ClickException(
                    "Either --held-out-cells or --auto-select must be provided"
                )

        manifest = GateManifest(
            parent_id=parent_id,
            created_at=datetime.now(timezone.utc).isoformat(),
            git_committed_at_sha="PENDING",
            held_out_cells=cells,
            K=K,
            p_threshold=p_threshold,
            bootstrap_reps=bootstrap_reps,
            win_definition=(
                "delta_composite > 0 AND p <= alpha/K (paired Wilcoxon, Bonferroni-corrected) "
                "AND ci_low > 0 (BCa bootstrap, 95%)"
            ),
            schema_version=SCHEMA_VERSION,
        )
        # Pre-write validation surfaces errors early
        import dataclasses
        validate_manifest_dict(dataclasses.asdict(manifest))

        manifests_dir = adir / "gate"
        sha = write_manifest_committed(manifest, manifests_dir, git_root)
        click.echo(
            f"Registered manifest for {parent_id} "
            f"(K={K}, p<{p_threshold}, held_out={len(cells)} cells, sha={sha[:12]})"
        )


    @gate_group.command("retire-manifest")
    @click.argument("parent_id")
    @click.option("--reason", required=True, help="Why this manifest is being superseded.")
    def retire_manifest_cmd(parent_id: str, reason: str) -> None:
        """Retire an active manifest (writes .retired.gate_manifest.json + git commit)."""
        from automil.cli._helpers import _find_automil_dir, _find_git_root
        from automil.gate import retire_manifest

        _validate_parent_id(parent_id)
        adir = _find_automil_dir()
        git_root = _find_git_root()
        manifests_dir = adir / "gate"
        sha = retire_manifest(parent_id, reason, manifests_dir, git_root)
        click.echo(f"Retired manifest for {parent_id} ({reason!r}, sha={sha[:12]})")


    @gate_group.command("status")
    @click.argument("parent_id", required=False)
    def gate_status_cmd(parent_id: str | None) -> None:
        """Show manifest details + per-candidate gate state."""
        from automil.cli._helpers import _find_automil_dir
        from automil.gate import load_manifest

        adir = _find_automil_dir()
        manifests_dir = adir / "gate"
        if not manifests_dir.exists():
            click.echo("No manifests registered.")
            return

        if parent_id:
            _validate_parent_id(parent_id)
            try:
                m = load_manifest(parent_id, manifests_dir)
            except FileNotFoundError as e:
                raise click.ClickException(str(e))
            click.echo(f"parent_id:           {m.parent_id}")
            click.echo(f"created_at:          {m.created_at}")
            click.echo(f"git_committed_at_sha:{m.git_committed_at_sha}")
            click.echo(f"K:                   {m.K}")
            click.echo(f"p_threshold:         {m.p_threshold}")
            click.echo(f"bootstrap_reps:      {m.bootstrap_reps}")
            click.echo(f"held_out_cells ({len(m.held_out_cells)}):")
            for cid, ds, enc, task in m.held_out_cells:
                click.echo(f"  {cid[:12]}  {ds:8s} {enc:12s} {task}")
        else:
            files = sorted(manifests_dir.glob("*.gate_manifest.json"))
            files = [f for f in files if not f.name.endswith(".retired.gate_manifest.json")]
            click.echo(f"Active manifests ({len(files)}):")
            for f in files:
                d = json.loads(f.read_text())
                click.echo(
                    f"  {d['parent_id']:12s}  K={d['K']:2d}  p={d['p_threshold']}  "
                    f"held_out={len(d['held_out_cells'])} cells"
                )


    @gate_group.command("stats")
    def gate_stats_cmd() -> None:
        """Show promotion_rate (30d window) + gate-health diagnostic (D-144 / GTE-06)."""
        from automil.cli._helpers import _find_automil_dir
        from automil.gate.stats import diagnose_gate_health
        from automil.graph import ExperimentGraph

        adir = _find_automil_dir()
        graph_path = adir / "graph.json"
        if not graph_path.exists():
            click.echo("No graph.json found.")
            return
        graph = ExperimentGraph(path=str(graph_path))
        rate_30 = graph.promotion_rate(days=30)
        nominated = len(graph.nominations_in_window(days=30))
        promoted = sum(1 for n in graph.nominations_in_window(days=30) if n.get("status") == "registered")
        click.echo(f"Promotion rate (30d): {rate_30:.1%} ({promoted}/{nominated})")
        click.echo(f"Health: {diagnose_gate_health(rate_30)}")
    ```
